Top10Tourna

select player.playerId,sum(matchdetail.wicket)
FROM MatchDetail,player,tournament,match
WHERE 
   matchdetail.playerId=player.playerId AND
   matchdetail.matchId=match.matchId AND
   match.tournamentId=tournament.tournamentId AND
   tournament.tournamentName=tname
GROUP by player.playerId
ORDER BY sum(matchdetail.wicket)
LIMIT 10
